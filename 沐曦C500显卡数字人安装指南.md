## 导出方法（纯离线）

从已安装的包生成 `.whl` 文件，**不联网**。

### 方法一：从缓存复制已有的 .whl

```bash
mkdir -p /data/shu/00_wheels && cd /data/shu/00_wheels

# 从 pip 缓存复制已有的 .whl
find /root/.cache/pip /opt/conda/pip/cache /tmp/pip -name "*.whl" -exec cp {} /data/shu/00_wheels \; 2>/dev/null

# 从 /mnt/wheel/ 复制（沐曦 MACA 包的源目录）
[ -d /mnt/wheel ] && cp /mnt/wheel/*.whl /data/shu/00_wheels/ 2>/dev/null

echo "已复制 $(ls *.whl 2>/dev/null | wc -l) 个 .whl 文件"
```

### 方法二：从已安装的包生成 .whl（纯离线，无需缓存）

如果缓存中已有的 `.whl` 不全，可以用以下 Python 脚本从已安装的包重建 `.whl`。

将以下内容保存为 `/data/shu/00_wheels/gen_wheels.py`：

```python
#!/usr/bin/env python3
"""从已安装的 Python 包离线生成 .whl 文件"""
import os, sys, csv, zipfile, json, email.parser, re, shutil, tempfile, sysconfig
from pathlib import Path

def find_site_packages():
    candidates = []
    for p in sys.path:
        if 'site-packages' in p and Path(p).exists():
            candidates.append(Path(p))
    if not candidates:
        import site
        candidates = [Path(p) for p in site.getsitepackages() if Path(p).exists()]
    return candidates[0] if candidates else None

def read_metadata(dist_info_dir):
    meta_path = dist_info_dir / 'METADATA'
    if not meta_path.exists():
        for f in dist_info_dir.iterdir():
            if f.name.endswith('.dist-info') and (f / 'METADATA').exists():
                return read_metadata(f)
        return None, None
    with open(meta_path, 'r', encoding='utf-8') as f:
        msg = email.parser.HeaderParser().parse(f)
    return msg.get('Name'), msg.get('Version')

def has_c_ext(sp_dir, top_level):
    for mod in top_level:
        p = sp_dir / mod
        if p.is_dir():
            if list(p.rglob('*.so')) or list(p.rglob('*.pyd')):
                return True
        elif p.is_file() and p.suffix in ('.so', '.pyd'):
            return True
    return False

def get_tags(sp_dir, top_level):
    py_ver = f"cp{sys.version_info.major}{sys.version_info.minor}"
    if has_c_ext(sp_dir, top_level):
        plat = sysconfig.get_platform().replace('-', '_').replace('.', '_')
        return (py_ver, py_ver, plat)
    return ('py3', 'none', 'any')

def wheel_filename(name, version, py_tag, abi, plat):
    return f"{name.replace('-', '_')}-{version}-{py_tag}-{abi}-{plat}.whl"

def collect_files_from_record(sp_dir, dist_info_dir):
    record_path = dist_info_dir / 'RECORD'
    files = []
    if record_path.exists():
        with open(record_path, 'r', encoding='utf-8') as f:
            for row in csv.reader(f):
                if row and row[0]:
                    full = (sp_dir / row[0]).resolve()
                    if full.exists() and str(full).startswith(str(sp_dir.resolve())):
                        files.append((row[0], full))
        if files:
            return files, True

    top_level = []
    tl_path = dist_info_dir / 'top_level.txt'
    if tl_path.exists():
        top_level = [l.strip() for l in open(tl_path) if l.strip()]

    for mod in top_level:
        p = sp_dir / mod
        if p.is_dir():
            for f in p.rglob('*'):
                if f.is_file():
                    rel = f.relative_to(sp_dir)
                    files.append((str(rel.as_posix()), f))
        elif p.is_file():
            files.append((mod, p))
        elif (sp_dir / f"{mod}.py").exists():
            files.append((f"{mod}.py", sp_dir / f"{mod}.py"))

    for f in dist_info_dir.iterdir():
        rel = f.relative_to(sp_dir)
        files.append((str(rel.as_posix()), f))

    return files, False

def gen_wheel(sp_dir, dist_info_dir, output_dir):
    pkg_name, pkg_ver = read_metadata(dist_info_dir)
    if not pkg_name or not pkg_ver:
        return None

    top_level = []
    tl_path = dist_info_dir / 'top_level.txt'
    if tl_path.exists():
        top_level = [l.strip() for l in open(tl_path) if l.strip()]

    files, used_record = collect_files_from_record(sp_dir, dist_info_dir)
    if not files:
        return None

    py_tag, abi, plat = get_tags(sp_dir, top_level)
    name = wheel_filename(pkg_name, pkg_ver, py_tag, abi, plat)
    out_path = output_dir / name
    if out_path.exists():
        return name

    with zipfile.ZipFile(out_path, 'w', zipfile.ZIP_DEFLATED) as zf:
        for rel_path, full_path in files:
            zf.write(full_path, rel_path)
    return name

def main():
    output_dir = Path(os.getcwd())
    sp_dir = find_site_packages()
    if not sp_dir:
        print("错误: 找不到 site-packages 目录")
        sys.exit(1)

    print(f"site-packages: {sp_dir}")
    print(f"输出目录: {output_dir}")
    print(f"{'='*60}")

    success, fail, skip = [], [], []
    for d in sorted(sp_dir.iterdir()):
        if not d.name.endswith('.dist-info'):
            continue
        name_ver = d.name.replace('.dist-info', '')
        out_name = f"{name_ver.replace('-', '_')}".split('-')
        pkg_base = name_ver.rsplit('-', 1)[0].replace('-', '_')

        existing = list(output_dir.glob(f"{pkg_base}-*.whl"))
        if existing:
            skip.append(existing[0].name)
            continue

        try:
            result = gen_wheel(sp_dir, d, output_dir)
            if result:
                success.append(result)
                print(f"  ✓ {result}")
            else:
                fail.append(d.name)
                print(f"  ✗ {d.name} (gen_wheel 返回 None)")
        except Exception as e:
            fail.append(d.name)
            print(f"  ✗ {d.name} ({e})")

    print(f"\n{'='*60}")
    print(f"  成功: {len(success)} 个")
    print(f"  失败: {len(fail)} 个")
    print(f"  已存在跳过: {len(skip)} 个")
    print(f"{'='*60}")

    if fail:
        print(f"\n失败的包（{len(fail)} 个）:")
        for f in fail:
            print(f"  {f}")

if __name__ == '__main__':
    main()
```
